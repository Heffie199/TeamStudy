--oracle删除序列化：

 DROP SEQUENCE seq_human;

--oracle创建序列化：
create sequence seq_human   
    minvalue 1   
    maxvalue 999999   
    start with 1   
    increment by 1   
    cache 20   
    cycle   
    order;  
   --获取sequence 获取主键 
 select seq_human.nextval from dual
